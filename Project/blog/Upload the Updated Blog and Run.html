the config in blog is:

# 未带www的请求，统一分发到https://www上
server {
    listen 80;
    server_name  blog.deep-diary.com;
    return 301 https://blog.deep-diary.com$request_uri;
}

# https 请求处理
server {
    listen 443 default_server ssl;
    server_name  blog.deep-diary.com;
    ssl_certificate /etc/nginx/ssh/8876673_blog.deep-diary.com.pem;
    ssl_certificate_key /etc/nginx/ssh/8876673_blog.deep-diary.com.key;

    large_client_header_buffers 4 16k;
    client_max_body_size 30m;
    client_body_buffer_size 128k;

    # 域名默认映射到 http://127.0.0.1:7777
    location / {
        proxy_pass http://127.0.0.1:7777;
        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        root html;
        index index.html index.html;

    }
}

Solution

in the serve, visit the dist file through nginx directly, so we need do some modification, in a nut shell:

==change the absolution path to relative path, / will be the root of the drive, instead of the project root. this is the reason that the project couldn't find the static resource==

update the nginx

server {
  charset utf-8;
  listen 80;
  server_name deepdiary.cn;  # 改成你的 IP

  # 定义 server 的根路径
  # 修改为你的项目的路径
  root /home/blog;


  # 其他所有请求均直接请求 Vue 打包的 html 文件
  location / {
    try_files /index.html =404;
  }
}
