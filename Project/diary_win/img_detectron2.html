安装cuda

下载 CUDA

去 NVIDIA 下载 CUDA，下载地址为：https://developer.nvidia.com/cuda-downloads (opens new window)。本次使用的是 Windows10 64位，因此此处选择 Windows10 64位版本，为了安装方便选择 local 版本。默认的下载是最新版本的 CUDA

若要下载其他版本的 CUDA，可以点击下面的 Legacy Releases (opens new window)，下载自己需要的 CUDA：

11.8

下载 cuDNN，下载之前需要先注册一下 Nvidia 的账号，下载地址为：[cudnn](cuDNN Download | NVIDIA Developer (opens new window)) , 根据上面的对应关系，此处下载8.9.1版本：

  • cuDNN中bin目录下的文件移动到 CUDA 的 bin 目录中

简单实用

import time

import torch
from detectron2.engine import DefaultPredictor
from detectron2.config import get_cfg
from detectron2.data import MetadataCatalog
from detectron2.utils.visualizer import Visualizer, ColorMode
from detectron2 import model_zoo
import cv2
import os
import numpy as np


class Detector:
    def __init__(self, model_type="OD", output_fold='../../output/img/'):
        self.cfg = get_cfg()
        self.model_type = model_type
        self.output_fold = output_fold

        # Load model config and pretrained model
        if model_type == "OD":  # Object Detection
            self.cfg.merge_from_file(model_zoo.get_config_file("COCO-Detection/faster_rcnn_R_101_FPN_3x.yaml"))
            self.cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url("COCO-Detection/faster_rcnn_R_101_FPN_3x.yaml")
        elif model_type == "IS":  # Instance Segmentation
            self.cfg.merge_from_file(model_zoo.get_config_file("COCO-InstanceSegmentation/mask_rcnn_R_101_FPN_3x.yaml"))
            self.cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url(
                "COCO-InstanceSegmentation/mask_rcnn_R_101_FPN_3x.yaml")
        elif model_type == "KP":  # keypoint detection
            self.cfg.merge_from_file(model_zoo.get_config_file("COCO-Keypoints/keypoint_rcnn_R_101_FPN_3x.yaml"))
            self.cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url(
                "COCO-Keypoints/keypoint_rcnn_R_101_FPN_3x.yaml")
        elif model_type == "LVIS":  # LVIS Segmentation
            self.cfg.merge_from_file(model_zoo.get_config_file("LVISv0.5-InstanceSegmentation/mask_rcnn_X_101_32x8d_FPN_1x.yaml"))
            self.cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url(
                "LVISv0.5-InstanceSegmentation/mask_rcnn_X_101_32x8d_FPN_1x.yaml")
        elif model_type == "PS":  # Panoptic Segmentation
            self.cfg.merge_from_file(model_zoo.get_config_file("COCO-PanopticSegmentation/panoptic_fpn_R_101_3x.yaml"))
            self.cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url(
                "COCO-PanopticSegmentation/panoptic_fpn_R_101_3x.yaml")

        self.cfg.MODEL.ROI_HEADS.SCORE_THRESH_TEST = 0.7  # set threshold for this model
        self.cfg.MODEL.DEVICE = 'cuda' if torch.cuda.is_available() else 'cpu'
        print("Using device: ", self.cfg.MODEL.DEVICE)

        self.predictor = DefaultPredictor(self.cfg)

    def imgProcess(self, img):
        if self.model_type != "PS":
            predictions = self.predictor(img)
            viz = Visualizer(img[:, :, ::-1], metadata=MetadataCatalog.get(self.cfg.DATASETS.TRAIN[0]),
                             instance_mode=ColorMode.IMAGE)
            output = viz.draw_instance_predictions(predictions["instances"].to("cpu"))
        else:
            panoptic_seg, segments_info = self.predictor(img)["panoptic_seg"]
            v = Visualizer(img[:, :, ::-1], MetadataCatalog.get(self.cfg.DATASETS.TRAIN[0]), scale=1.2)
            output = v.draw_panoptic_seg_predictions(panoptic_seg.to("cpu"), segments_info)
        img = output.get_image()[:, :, ::-1]
        return img

    def onImage(self, imagePath, resize_rat=1.0):
        image = cv2.imread(imagePath)
        #  保持原比例改变大小
        image = cv2.resize(image, (int(image.shape[1] * resize_rat), int(image.shape[0]  * resize_rat)))
        output = self.imgProcess(image)
        self.onSaveImg(output, name=os.path.basename(imagePath))
        # cv2.imshow("Result", output)
        # cv2.waitKey(0)

    def onVideo(self, videoPath):
        cap = cv2.VideoCapture(videoPath)
        if not cap.isOpened():
            print("Error opening video stream or file")
            return
        ret, image = cap.read()
        while ret:
            output = self.imgProcess(image)
            cv2.imshow("Result", output)
            key = cv2.waitKey(1) & 0xFF
            if key == ord("q"):
                break
            ret, image = cap.read()
        cap.release()
        cv2.destroyAllWindows()

    #  save image, which has been detected and marked
    def onSaveImg(self, img, name='t1_output.jpg'):
        """
        :param img: the opened image
        :param name: the saved image name
        :return:
        """
        cv2.imwrite(os.path.join(self.output_fold, name), img)

    def __creatOutFold__(self):
        ##creat a new saving flod
        current_time = time.localtime()
        self.output_fold = os.path.join(self.output_fold,
                                     'img_detectron2' + '_' + time.strftime("%Y_%m_%d_%H_%M_%S", current_time))
        os.makedirs(self.output_fold, exist_ok=True)
        return self.output_fold

output_fold = detector.__creatOutFold__()
