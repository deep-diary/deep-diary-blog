简单实用

注意： insight face 识别出来后，截取出来的人脸无法再次进行识别

import os
import random
import shutil
import string
from os import path
import os as osp
import pickle
import time

import cv2 as cv
import numpy as np
from insightface.app import FaceAnalysis


class ImgInsight:

    def __init__(self, det_thresh=0.5, output_fold='../../output/img/'):

        self.run_type = 'img_insight'
        self.output_fold = output_fold
        self.ft_targe = None  # store the target face feature, if not None, then compare with this feature
        self.app = FaceAnalysis(
            providers=['CUDAExecutionProvider', 'CPUExecutionProvider'])  # allowed_modules=['detection']
        self.app.prepare(ctx_id=0, det_size=(640, 640), det_thresh=det_thresh)

    #  do the face recognition from the image
    def onImage(self, image_file, resize_rat=1.0, to_rgb=False):
        """
        :param image_file: the image file path or the image data that need to be processed
        :param resize_rat: resize the image before process
        :param to_rgb: change the image from bgr to rgb
        :return img: image data
        :return faces: all faces information in the image
        """
        if type(image_file) is str:
            img = cv.imread(image_file)
        else:
            img = image_file
        img = cv.resize(img, (int(img.shape[1] * resize_rat), int(img.shape[0] * resize_rat)))

        if to_rgb:
            img = img[:, :, ::-1]
        start = time.time()
        faces = self.app.get(img)
        end = time.time()
        print('get faces time: ', end - start)
        return img, faces

    # do the face recognition from the fold
    def onFold(self, fold_path, recurve=False, save_img=False, save_face=False, **params):
        """
        :param fold_path: the image fold path
        :param recurve: if True, then recurve the sub-fold
        :param save_img: if True, same the image with marked face
        :param save_face: if True, save the face as an image
        :param params: other params
        :return:
        """

        img_nums = 0
        face_nums = 0
        feats = []
        name = osp.path.basename(fold_path)  # flod name
        # name = flod_path.split(osp.path.sep)[-1]   # the same with above line
        imagePaths = [osp.path.join(fold_path, f) for f in osp.listdir(fold_path)]
        for (i, imagePath) in enumerate(imagePaths):
            # extract the person name from the image path
            print("[INFO] processing image {}/{}".format(i + 1, len(imagePaths)))
            #  判断文件是否为图片
            if not imagePath.lower().endswith(
                    ('.bmp', '.dib', '.png', '.jpg', '.jpeg', '.pbm', '.pgm', '.ppm', '.tif', '.tiff')):
                print('请选择正确的图片格式')
                continue

            img, faces = self.onImage(imagePath)
            if len(faces) > 0:
                if save_img:
                    print('save image---------------------------------------------------------------')
                    self.onSaveImg(img, faces, name=path.basename(imagePath))  # 保存图片, 按原来文件名命名
                if save_face:
                    self.onSaveFace(img, faces, name=path.basename(imagePath))  # 保存人脸, 按原来文件名命名

            [feats.append(fc.normed_embedding) for fc in faces]

        img_nums = len(imagePaths)
        face_nums = len(feats)
        print(f'共有{img_nums}张图片，检测到{face_nums}张人脸')

        return name, feats

    ## 从整个人脸目录中获取人脸数据
    # input:
    #   flod_path       照片目录
    #   **params        相关配置参数
    # output
    #   imgs            目录中所有照片
    #   faces           目录中所有照片对应人脸
    def from_train_flod(self, flod_path, **params):

        # Grab the paths to the input images in our dataset
        print("[INFO] quantifying faces...")
        namePaths = osp.listdir(flod_path)

        # Initialize our lists of extracted facial embeddings and corresponding people names
        knownEmbeddings = []
        knownNames = []
        knownClasify = []

        # Initialize the total number of faces processed
        total = 0

        # Loop over the imagePaths
        for (i, namePath) in enumerate(namePaths):
            # extract the person name from the image path
            print("[INFO] processing name {}/{}".format(i + 1, len(namePaths)))
            name, feats = self.onFold(osp.path.join(flod_path, namePath))

            if len(feats) == 0:
                continue
            # add the name of the person + corresponding face
            # embedding to their respective list
            knownNames.append(name)  # 一个文件夹是一个人脸，但一个文件夹中可以训练多张照片
            for feat in feats:
                knownEmbeddings.append(feat)
                knownClasify.append(i)
            total += 1

        print(total, " names embedded")

        # save to output
        data = {"embeddings": knownEmbeddings, "names": knownNames, "classify": knownClasify}
        self.knownNames = knownNames
        self.knownClasify = knownClasify
        f = open(self.embedding_path, "wb")
        f.write(pickle.dumps(data))
        f.close()

    # draw the face and landmark on the image
    def onDraw(self, img, faces):
        """
        :param img: the opened image
        :param faces: faces information
        :return: 
        """
        dimg = img.copy()
        for i in range(len(faces)):
            face = faces[i]
            box = face.bbox.astype(np.int32)
            color = (0, 0, 255)
            cv.rectangle(dimg, (box[0], box[1]), (box[2], box[3]), color, 2)
            if face.kps is not None:
                kps = face.kps.astype(np.int32)
                # print(landmark.shape)
                for l in range(kps.shape[0]):
                    color = (0, 0, 255)
                    if l == 0 or l == 3:
                        color = (0, 255, 0)
                    cv.circle(dimg, (kps[l][0], kps[l][1]), 1, color,
                               2)
            if face.gender is not None and face.age is not None:
                cv.putText(dimg, '%s,%d' % (face.sex, face.age), (box[0] - 1, box[1] - 4), cv.FONT_HERSHEY_COMPLEX,
                            0.7, (0, 255, 0), 1)

            for key, value in face.items():
                if key.startswith('landmark_3d'):
                    print(key, value.shape)
                    # print(value[0:10, :])
                    lmk = np.round(value).astype(np.int32)
                    for l in range(lmk.shape[0]):
                        color = (255, 0, 0)
                        cv.circle(dimg, (lmk[l][0], lmk[l][1]), 1, color,
                                   2)
        return dimg

    #  save image, which has been detected and marked
    def onSaveImg(self, img, faces, name='t1_output.jpg'):
        """
        :param img: the opened image
        :param faces: faces information
        :param name: the saved image name
        :return:
        """
        # if faces:
        rimg = imgInsight.onDraw(img, faces)
        cv.imwrite(path.join(self.output_fold, name), rimg)

    #  save face, which has been detected
    def onSaveFace(self, img, faces, name='unknown'):
        """
        :param img: original image
        :param faces: all the face information in the image
        :param name: image name
        :return:
        """
        for face in faces:
            print(f'----det_score: {face.det_score}-----')

            # judge whether the face folod is exists
            fcname = name.split(".")[0] + '_' + 'face_' + ''.join(
                (random.sample(string.ascii_letters + string.digits, 5))) + '.jpg'  # face name with random letters
            face_path = path.join(self.output_fold, fcname)  # face path
            face_box = face.bbox.astype(int)
            if face_box.min() < 0:  # 不完整的人脸
                return
            # zoom the face box area by ratio to make the face area bigger
            # face_box formate is [x1, y1, x2, y2], which is the left-top and right-bottom point
            face_box = self.onZoomFaceBox(face_box, img.shape, ratio=0.2)

            face_img = img[face_box[1]:face_box[3], face_box[0]:face_box[2]]
            if self.ft_targe is not None:
                ft = face.normed_embedding
                sim = np.dot(ft, self.ft_targe)
                print('similarity: ', sim)
                if sim > 0.4:
                    cv.imwrite(face_path, face_img)
            else:
                cv.imwrite(face_path, face_img)

    def onZoomFaceBox(self, face_box, shape, ratio=0.2):
        """
        :param face_box: the face box area
        :param shape: the shape of the image, format is [h, w, c]
        :param ratio: the ratio to zoom the face box area
        :return: the zoomed face box area
        """
        x1, y1, x2, y2 = face_box
        w = x2 - x1
        h = y2 - y1
        x1 = x1 - int(w * ratio)
        x2 = x2 + int(w * ratio)
        y1 = y1 - int(h * ratio)
        y2 = y2 + int(h * ratio)
        # do a limit for the face box area
        x1 = max(x1, 0)
        x2 = min(x2, shape[1])
        y1 = max(y1, 0)
        y2 = min(y2, shape[0])

        return [x1, y1, x2, y2]

    # save the specified face.
    def onSaveSpecifiedFace(self, fcs_fold, target_fold, fc_path):
        """
        :param fcs_fold: which fold include all the faces
        :param target_fold: which is used to save the specified face only
        :param fc_path: the specified face path
        :return: none
        """
        img = cv.imread(fc_path)
        fc = self.app.get(img)
        ft_target = fc[0].normed_embedding
        # brow all the faces in the fcs_fold
        for fc_name in os.listdir(fcs_fold):
            fc_path = path.join(fcs_fold, fc_name)
            img = cv.imread(fc_path)
            fc = self.app.get(img)
            if fc:
                print('------face detected')
                ft = fc[0].normed_embedding
                sim = np.dot(ft, ft_target)
                print('similarity: ', sim)

                face_box = fc[0].bbox.astype(int)
                if face_box.min() < 0:  # 不完整的人脸
                    return
                face_box = self.onZoomFaceBox(face_box, img.shape, ratio=0.2)
                face_img = img[face_box[1]:face_box[3], face_box[0]:face_box[2]]

                if sim > 0.4:
                    # shutil.copy(fc_path, path.join(target_fold, fc_name))  # copy the face to the target fold
                    cv.imwrite(path.join(target_fold, fc_name), face_img)  # save the face to the target fold

    def __creatOutFold__(self):
        ##creat a new saving flod
        current_time = time.localtime()
        self.output_fold = path.join(self.output_fold,
                                     'img_insight' + '_' + time.strftime("%Y_%m_%d_%H_%M_%S", current_time))
        os.makedirs(self.output_fold, exist_ok=True)
        return self.output_fold

这里主要可以设置2个参数：

  • 大图宽度
  • 宽高比例

原理

根据图片数量，计算小图宽度及大图列数

就是通过动态规划算法，指定人脸存放路径，自动生成大拼图

动态规划算法

所谓动态规划算法，就是加入我们目标是100，现在有好几张图片的长度为 len = [23, 45, 76, 12, 7, .......]

可以理解为，从len数组中，抽取出若干个数，其综合正好等于100

难点

就是对照片没有进行裁剪，但需要用到全部的照片，并正好铺满整个画布

代码实现

文件夹中的照片，尺寸约不一样越好，如果大部分长宽比都一样，会导致无法实现预期的效果

import os
import time

import PIL.Image as Image
import numpy as np


class ImgCanvas:

    def __init__(self, layout='horizontal'):
        """
        :param layout: the layout of big image, or called canvas, default is horizontal
        :return:
        """
        self.layout = layout
        self.output_fold = '../../output/img/'

    # get all the faces in the folder and resize the faces
    def image_process(self, fcs_fold, targe_wid=1080, target_aspect=0.5625):
        """
        :param fcs_fold: face folder, which includes all the faces
        :param targe_wid: targe big image width
        :param target_aspect: targe aspect ratio, default is 0.5625, which means wid:height = 9/16
        :return image_files: all the faces file names
        :return resized_images_height: all the faces height
        :return resized_images: all the faces
        :return collage_width: the big image's width
        :return collage_height: the big image's height
        :return step: how many steps to fill all the small image, could be understood as how many column
        :return fixed_len: Calculated the width of the small image
        """

        # judge whehter the fcs_fold is a valid folder
        if not os.path.isdir(fcs_fold):
            raise ValueError("The input folder is not valid!")

        # 获取文件夹中的所有图片文件
        image_files = [f for f in os.listdir(fcs_fold) if f.endswith(('.jpg', '.jpeg', '.png'))]
        tol_imgs = len(image_files)
        if tol_imgs < 20:
            print(
                f'the fold have too little images, it will be hard to combine to a big image, pld try add more images')
            return
        print(f'INFO: total found {tol_imgs}')

        tol_wid = 0
        tol_height = 0
        # 计算出所有图片的总长宽比ratio
        for image_file in image_files:
            image_path = os.path.join(fcs_fold, image_file)
            with Image.open(image_path) as image:
                width, height = image.size
                tol_wid += width
                tol_height += height

        ratio = tol_height / tol_wid
        print(f'INFO: total ratio is  {ratio}')

        # 计算大图列数，大图默认target_aspect为9：16
        step = int(pow(target_aspect * tol_imgs * ratio, 0.5))
        print(f'INFO: total cols is  {step}')

        # 由于整个大图一共有step列，fixed_len = targe_wid // step
        fixed_len = int(targe_wid // step)
        print(f'INFO: fixed_len is {fixed_len}')

        # 将小照片的宽度都等比例调整到fixed_len
        resized_images = []
        resized_images_height = []
        for image_file in image_files:
            image_path = os.path.join(fcs_fold, image_file)
            with Image.open(image_path) as image:
                width, height = image.size
                new_width = fixed_len
                new_height = int(height * (fixed_len / width))
                resized_image = image.resize((new_width, new_height), Image.LANCZOS)
                resized_images.append(resized_image)
                resized_images_height.append(new_height)

        # 计算调整好宽度后所有照片的总长度len_total
        # len_total = sum([image.size[1] for image in resized_images])
        len_total = sum(resized_images_height)

        # 由于整个大图一共有step列，计算单列的长度len = len_total/fixed_len
        single_col_len = len_total // step

        # 计算大图的理论宽度和高度, 宽度=fixed_len*step, 高度=len
        collage_width = fixed_len * step
        collage_height = single_col_len

        return image_files, resized_images_height, resized_images, collage_width, collage_height, step, fixed_len

    # dynamic programming algorithm
    def dp_opt(self, total_space, parts):
        """
        :param total_space: the total space
        :param parts: all the parts array, each parts include some space
        :return dp: histor result array
        :return i: parts idx that could get the results
        :return j: space idx that could get the results
        :return rest_space: if tried everything, the left space
        """
        w_len = len(parts)
        dp = [[0] * (total_space + 1) for _ in range(w_len + 1)]
        i = 0
        j = 0
        for i in range(1, w_len + 1):  # parts loop
            for j in range(1, total_space + 1):  # space loop
                if parts[i - 1] <= j:
                    for k in range(0, i):  # history parts loop
                        dp[i][j] = max(max(dp[k][j - parts[i - 1]] + parts[i - 1], dp[i][j]), dp[i - 1][j])
                        if dp[i][j] == total_space:  # finish condition

                            return dp, i, j, total_space - dp[i][j]
                else:
                    dp[i][j] = dp[i - 1][j]
        return dp, i, j, total_space - dp[i][j]

    # trace back the dp result
    def dp_opt_trace_back(self, dp, i, j, rest_space, w):
        """
        :param dp: history result array
        :param i: parts idx that could get the results
        :param j: space idx that could get the results
        :param rest_space: if tried everything, the left space
        :param w: each parts space
        :return rest_values: parts that haven't used
        :return used_values: parts that already used
        :return used_idx: parts idx that already used
        """
        w_idx = []
        j = j - rest_space  # reduce the rest space
        while dp[i][j] != 0:  # trace back
            rest_w = j - w[i - 1]
            is_change = False
            for k in range(0, i):
                if dp[k][rest_w] == rest_w:
                    w_idx.append(i - 1)
                    print(f'used weight is :{w_idx}')
                    i = k
                    j = rest_w
                    is_change = True
                    break
            if not is_change:
                print('could not find the used weight, the reason is that the source data is not enough, pls increase '
                      'the face numbers or reduce the step numbers and try again!')
                break
                # exit(0)

        # rest_values = np.delete(w, w_idx)
        rest_values = w
        for i in range(len(w_idx)):
            rest_values[w_idx[i]] = 0

        used_values = np.array(w)[w_idx]
        used_idx = w_idx
        return rest_values, used_values, used_idx

    def on_create_collage(self, fcs_fold, targe_wid=1080, target_aspect=0.5625):
        """
        :param fcs_fold: face folder, which includes all the faces
        :param targe_wid: targe big image width
        :param target_aspect: targe aspect ratio, default is 0.5625, which means wid:height = 9/16
        """
        # 获取图片集地址下的所有图片名称
        image_names, image_height, resized_images, collage_width, collage_height, step, fixed_len = self.image_process(
            fcs_fold, targe_wid=targe_wid, target_aspect=target_aspect)
        to_image = Image.new('RGB', (collage_width, collage_height))  # 创建一个新图
        w_pt = 0
        h_pt = 0

        total_space = collage_height
        rest_values = image_height.copy()
        used_idx = []
        used_img = 0
        print(
            f'INFO: image_maker--->the target image is  {collage_width}, {collage_height} , total space is {total_space}')
        # 动态规划
        for stp in range(step):
            dp, i, j, rest_space = self.dp_opt(total_space, rest_values)
            rest_values, used_values, idx = self.dp_opt_trace_back(dp, i, j, rest_space, rest_values)
            used_idx.append(idx)

            # 循环遍历，把每张图片按顺序粘贴到对应位置上
            for img_idx in idx:
                to_image.paste(resized_images[img_idx], (w_pt, h_pt))
                h_pt += image_height[img_idx]
                used_img += 1
            # 补偿最后一张照片的误差
            if rest_space:  # and stp == step-1
                img_idx = [i for i, e in enumerate(rest_values) if e != 0]
                print(
                    f'---there is no more suitable size yet, rest_space is {rest_space}, while rest_values is {rest_values[img_idx[0]]}-----')
                print(f'---total height is {collage_height}, h_pt is {h_pt}-----')
                to_image.paste(resized_images[img_idx[0]], (w_pt, h_pt))
                used_img += 1
            print(f'INFO: ----------col {stp} finished, there are {used_img} have been used--------')

            w_pt += fixed_len  # move the point to another column
            h_pt = 0
        to_image.show()

        outpath = os.path.join(self.__creatOutFold__(), 'face_maker_out.jpg')
        to_image.save(outpath)  # 保存新图
        return outpath  # 新图路径

    def __creatOutFold__(self):
        # creat a new saving fold
        current_time = time.localtime()
        self.output_fold = os.path.join(self.output_fold,
                                        'img_maker' + '_' + time.strftime("%Y_%m_%d_%H_%M_%S", current_time))
        os.makedirs(self.output_fold, exist_ok=True)
        return self.output_fold
