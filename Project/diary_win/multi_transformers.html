import requests
from PIL import Image
import os
from io import BytesIO
import tempfile

def open_image(img_path, is_save=True):
    """
    :param img_path: could be URL or local path
    :param is_save: if the img_path is URL, then save the image to local path
    :return image: the opened image
    :return path: local image path or Temporary path if the img_path is URL; None if is_save is False
    """
    path = None
    if img_path.startswith('http://') or img_path.startswith('https://'):
        response = requests.get(img_path)
        image = Image.open(BytesIO(response.content))

        if is_save:
            with tempfile.NamedTemporaryFile(delete=False) as temp_file:
                temp_file.write(response.content)
                path = temp_file.name

    else:  # Local path
        if os.path.isfile(img_path):
            image = Image.open(img_path)
            if is_save:
                path = img_path
        else:
            raise FileNotFoundError(f"Cannot find the image file at '{img_path}'")

    return image, path
